# WhisperX brings in faster-whisper, pyannote.audio, torch, and torchaudio.
# 3.3.x resolves to wheels on Apple Silicon (av>=17, faster-whisper>=1.1,
# pyannote.audio 3.x) — avoiding the PyAV-from-source build that older
# whisperx (==3.1.5 -> av==11, no arm64 wheel) triggered. It keeps the classic
# load_model / DiarizationPipeline / align / assign_word_speakers API.
whisperx==3.3.1
# Pin the ML stack to a coherent set whisperx 3.3.x / pyannote.audio 3.3.x were
# built against. Without these, pip pulls the newest torch/torchaudio (2.11+,
# which removed torchaudio.AudioMetaData that pyannote.audio 3.3.2 still uses)
# and transformers 5.x (post-dates whisperx 3.3) — both break at import time.
torch==2.5.1
torchaudio==2.5.1
transformers<5
# pyannote.audio 3.3.2 imports matplotlib but doesn't declare it as a dep.
matplotlib
fastapi==0.115.6
uvicorn[standard]==0.34.0
python-multipart==0.0.20
